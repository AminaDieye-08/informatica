// Funzione principale function calcolaRisultati(a, b, c, d) { // Primo risultato: somma pesata degli input const primoRisultato = (a * 0.5) + (b * 0.3) + (c * 0.1) + (d * 0.1); // Secondo risultato: applicazione della funzione sigmoide al primo risultato const secondoRisultato = 1 / (1 + Math.exp(-primoRisultato)); // Restituisce entrambi i risultati return { primoRisultato, secondoRisultato }; } // Test della funzione const input = { a: 2, b: 3, c: 1, d: 4 }; const risultati = calcolaRisultati(input.a, input.b, input.c, input.d); console.log("Primo risultato:", risultati.primoRisultato); console.log("Secondo risultato:", risultati.secondoRisultato);
