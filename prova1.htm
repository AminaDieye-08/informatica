// Funzione per calcolare due output basati sugli input numerici function calcolaOutput(a, b, c, d) { // Primo output: media dei quattro numeri const primoOutput = (a + b + c + d) / 4; // Secondo output: quadrato del primo output const secondoOutput = Math.pow(primoOutput, 2); // Restituzione dei risultati return { primoOutput, secondoOutput }; } // Esempio di utilizzo della funzione const risultati = calcolaOutput(2, 4, 6, 8); console.log("Primo Output:", risultati.primoOutput); // Risultato: 5 console.log("Secondo Output:", risultati.secondoOutput); // Risultato: 25
